/* Nao consegui fazer o MySQL/MariaDB aceitar o uso do comando "system mysql -u <user> -p" sem pedir senha, (comando necessario para se conectar com outro usuario) */
/* mesmo que no "CREATE USER" eu nao usasse o termo "IDENTIFIED BY", ele pede a senha mesmo nao existindo, obviamente dando erro. E se tentar rodar o script inteiro junto com */
/* o comando de conexao, o MySQL pede senha e nao vai funcionar, ele nem te da a oportunidade de escrever a senha, e vai bugar todos os outros comandos que vem apos, */
/* entao criei um usuario com senha, e dividi meu script em dois, a fim de nao bugar todo meu codigo. SENHA DO USER: 123 */

DROP USER natan; -- Deletando o user natan caso ja exista.
DROP DATABASE uvv; -- Deletando o banco de dados uvv caso ja exista.

CREATE USER natan IDENTIFIED BY '123';  -- Criando o usuario natan e colocando a senha de 123.

GRANT CREATE, ALTER, DROP, REFERENCES, DELETE, UPDATE, INDEX, INSERT, SELECT ON uvv.* TO natan; -- Garantindo para o meu usuario os privilegios de criar, deletar, referenciar, atualizar, inserir e projetar no em todo o banco de dados uvv.

-- Conectando ao MySQL como o user natan.
-- ATENCAO !!! SENHA: 123
system mysql -u natan -p
